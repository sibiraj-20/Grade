#include<stdio.h>
int main ()
{
  int m1, m2, m3, m4, m5, total, avg;
  printf ("Enter the value of the a to e ");
  scanf ("%d", &m1);
  scanf ("%d", &m2);
  scanf ("%d", &m3);
  scanf ("%d", &m4);
  scanf ("%d", &m5);
  total = m1 + m2 + m3 + m4 + m5;
  avg = total / 5;
  printf ("The total is :%d", total);
  printf ("\nThe avg is :%d", avg);
  if (m1 >= 35 && m2 >= 35 && m3 >= 35 && m4 >= 35 && m5 >= 35)
    {
      printf ("\nResult : Pass");
	if (avg >= 90 && avg <= 100)
	  {
	    printf ("\n Grade : A Grade");
	  }
	else if (avg >= 80 && avg <= 89)
	  {
	    printf ("\n Grade : B Grade");
	  }
	else if (avg >= 70 && avg <= 79)
	  {
	    printf ("\nGrade : C Grade");
	  }
	else
	  {
	    printf ("\n Grade : D Grade");
	  }
     }
     else
      {
	printf ("\nResult : Fail");
      }
      return 0;
}
